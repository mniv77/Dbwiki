        // Load on page load
        window.onload = function() {
            loadStats();

        // ====================================================================
        // LOAD DATA
        // ====================================================================

        function loadStats() {
            fetch('/api/wiki/stats')
                .then(r => r.json())
                .then(data => {
                    if (data.success) {
                        document.getElementById('stat-total').textContent = data.stats.total_ideas || 0;
                        document.getElementById('stat-comments').textContent = data.stats.total_comments || 0;
                        document.getElementById('stat-recent').textContent = data.stats.recent_ideas || 0;
                        document.getElementById('stat-categories').textContent = data.stats.by_category?.length || 0;
                        
                        // Populate category dropdown
                        const categorySelect = document.getElementById('filter-category');
                        data.stats.by_category?.forEach(cat => {
                            const option = document.createElement('option');
                            option.value = cat.category;
                            option.textContent = `${cat.category} (${cat.count})`;
                            categorySelect.appendChild(option);
                        });
                    }
                });
        }

        function loadIdeas(filters = {}) {
            const params = new URLSearchParams(filters);
            fetch(`/api/wiki/ideas/list?${params}`)
                .then(r => r.json())
                .then(data => {
                    if (data.success) {
                        allIdeas = data.ideas;
                        displayIdeasList(data.ideas);
                    }
                });
        }

        // ====================================================================
        // DISPLAY IDEAS - Wiki Style (Title ‚Üí Summary ‚Üí Full)
        // ====================================================================

                `;
                return;
            }

            container.innerHTML = ideas.map(idea => `

                    <!-- LEVEL 1: TITLE (Always visible) -->

                            <div class="flex-1">
                                <h3 class="text-xl font-bold text-blue-400 mb-1">
                                    <span id="arrow-${idea.id}">‚ñ∂</span> ${escapeHtml(idea.idea_title)}

                            </p>

        // ====================================================================
        // TOGGLE SUMMARY (Title ‚Üí Summary)
        // ====================================================================

        function toggleSummary(ideaId) {
            const summaryDiv = document.getElementById(`summary-${ideaId}`);
            const arrow = document.getElementById(`arrow-${ideaId}`);
            
            if (summaryDiv.classList.contains('collapsed')) {
                summaryDiv.classList.remove('collapsed');
                summaryDiv.classList.add('expanded');
                arrow.textContent = '‚ñº';
            } else {
                summaryDiv.classList.add('collapsed');
                summaryDiv.classList.remove('expanded');
                arrow.textContent = '‚ñ∂';
            }
        }

            fetch(`/api/wiki/ideas/${ideaId}`)
                .then(r => r.json())
                .then(data => {
                    if (data.success) {
                        displayFullIdea(data.idea, data.comments);
                    }
                });
        }

        // ====================================================================
        // COMMENTS
        // ====================================================================

        function submitComment(ideaId) {
            const comment = document.getElementById(`new-comment-${ideaId}`).value.trim();
            const name = document.getElementById(`comment-name-${ideaId}`).value.trim() || 'Anonymous';
            const type = document.getElementById(`comment-type-${ideaId}`).value;
            
            if (!comment) {
                alert('Please enter a comment');
                return;
            }
            
            fetch('/api/wiki/comments', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    idea_id: ideaId,
                    comment_text: comment,
                    comment_type: type,
                    commenter_name: name
                })
            })
            .then(r => r.json())
            .then(data => {
                if (data.success) {
                    // Reload the full idea to show new comment
                    document.querySelector('.fixed').remove();
                    showFullIdea(ideaId);
                }
            });
        }

        function resolveComment(commentId, ideaId) {
            fetch(`/api/wiki/comments/${commentId}/resolve`, {
                method: 'PUT',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ resolved_by: 'Meir' })
            })
            .then(r => r.json())
            .then(data => {
                if (data.success) {
                    // Reload the full idea
                    document.querySelector('.fixed').remove();
                    showFullIdea(ideaId);
                }
            });
        }

        // ====================================================================
        // SEARCH & FILTER
        // ====================================================================

        function searchIdeas() {
            const filters = {
                search: document.getElementById('search-box').value,
                category: document.getElementById('filter-category').value,
                type: document.getElementById('filter-type').value
            };
            loadIdeas(filters);
        }

        function clearSearch() {
            document.getElementById('search-box').value = '';
            document.getElementById('filter-category').value = '';
            document.getElementById('filter-type').value = '';
            loadIdeas();
        }

        // ====================================================================
        // ADD NEW IDEA
        // ====================================================================

        function showAddForm() {
            document.getElementById('add-form').classList.remove('hidden');
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        function hideAddForm() {
            document.getElementById('add-form').classList.add('hidden');
            document.getElementById('new-idea-form').reset();
        }

        function submitIdea(e) {
            e.preventDefault();
            
            const data = {
                title: document.getElementById('form-title').value,
                summary: document.getElementById('form-summary').value,
                content: document.getElementById('form-content').value,
                type: document.getElementById('form-type').value,
                category: document.getElementById('form-category').value,
                priority: document.getElementById('form-priority').value,
                tags: document.getElementById('form-tags').value,
                source_file: document.getElementById('form-source').value
            };
            
            fetch('/api/wiki/ideas', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(data)
            })
            .then(r => r.json())
            .then(result => {
                if (result.success) {
                    hideAddForm();
                    loadStats();
                    loadIdeas();
                    alert('Idea added successfully!');
                }
            });
        }

        // ====================================================================
        // HELPER FUNCTIONS

        function getPriorityColor(priority) {
            const colors = {
                'Critical': 'border-red-500',
                'High': 'border-orange-500',
                'Medium': 'border-yellow-500',
                'Low': 'border-green-500'
            };
            return colors[priority] || 'border-gray-500';
        }

        function formatDate(dateStr) {
            const date = new Date(dateStr);
            return date.toLocaleDateString() + ' ' + date.toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'});
        }

        function escapeHtml(text) {
            if (!text) return '';
            const div = document.createElement('div');
            div.textContent = text;
            return div.innerHTML;
        }
